## Materi

### Konsep Dasar

Regular expressions (regex) adalah powerful tool untuk pattern matching dan text manipulation. Python menyediakan modul `re`.

**Import dan basic usage:**
```python
import re

text = "Alice email: alice@example.com, Bob: bob@test.org"

# search: cari pattern pertama yang match
match = re.search(r"\w+@\w+\.\w+", text)
if match:
    print(match.group())  # alice@example.com

# findall: semua matches sebagai list
emails = re.findall(r"\w+@\w+\.\w+", text)
# ['alice@example.com', 'bob@test.org']

# match: hanya di awal string
re.match(r"\w+", text)  # match karena text mulai dengan word
re.match(r"Bob", text)  # None (tidak di awal)
```

### Detail Teknis

**1. Raw Strings (penting!)**

```python
# Normal string - backslash di-interpret
pattern = "\d+"  # \d diubah jadi literal 'd' (tidak sesuai yang diharapkan)

# Raw string - backslash literal
pattern = r"\d+"  # benar: \d sebagai regex untuk digit

# Selalu gunakan raw strings untuk regex patterns
```

**2. Metacharacters & Special Sequences**
```python
# . (dot) - any character except newline
re.search(r"h.t", "hat")   # match
re.search(r"h.t", "hot")   # match
re.search(r"h.t", "h t")  # tidak match (space)

# ^ - start of string
re.search(r"^Hello", "Hello World")  # match
re.search(r"^Hello", "World Hello")  # no match

# $ - end of string
re.search(r"World$", "Hello World")  # match

# * - 0 or more repetitions
re.search(r"ab*c", "ac")     # match (0 b)
re.search(r"ab*c", "abc")    # match
re.search(r"ab*c", "abbbc")  # match

# + - 1 or more repetitions
re.search(r"ab+c", "abc")    # match
re.search(r"ab+c", "ac")    # no match (butuh minimal 1 b)

# ? - 0 or 1 repetition (optional)
re.search(r"colou?r", "color")   # match (u optional)
re.search(r"colou?r", "colour")  # match

# {n} - exactly n repetitions
re.search(r"\d{3}", "abc123")  # match "123"

# {n,m} - between n and m repetitions (inclusive)
re.search(r"\d{2,4}", "a12b")  # match "12"
re.search(r"\d{2,4}", "a1234b")  # match "1234"

# Character classes []
re.search(r"[aeiou]", "hello")  # match (ada vowel)
re.search(r"[0-9]", "abc")      # no match
re.search(r"[A-Za-z]", "123")   # no match

# Negated character classes [^...]
re.search(r"[^0-9]", "123")     # no match (harus non-digit)
re.search(r"[^aeiou]", "aei")   # no match

# Special sequences (shorthand)
re.search(r"\d", "a1b")     # digit (equiv [0-9])
re.search(r"\D", "123")     # non-digit (equiv [^0-9])
re.search(r"\w", "abc_123") # word char (equiv [a-zA-Z0-9_])
re.search(r"\W", "abc123")  # non-word char
re.search(r"\s", "a b")     # whitespace (space, tab, newline)
re.search(r"\S", " a ")     # non-whitespace

# Alternation |
re.search(r"cat|dog", "I have a cat")  # match "cat"
re.search(r"cat|dog", "I have a dog")  # match "dog"

# Grouping dengan ()
match = re.search(r"(\w+)@(\w+)\.(\w+)", "alice@example.com")
if match:
    print(match.group(0))  # alice@example.com (full match)
    print(match.group(1))  # alice (first group)
    print(match.group(2))  # example (second group)
    print(match.group(3))  # com (third group)

# Named groups (?P<name>...)
match = re.search(r"(?P<user>\w+)@(?P<domain>\w+)\.(?P<tld>\w+)", "bob@test.org")
if match:
    print(match.group("user"))   # bob
    print(match.group("domain")) # test
    print(match.group("tld"))    # org
```

**3. Greediness**
```python
# Greedy (default): match sebanyak mungkin
text = "<div>content</div>"
re.search(r"<.*>", text).group()  # "<div>content</div>" (selamanya sampai akhir)

# Non-greedy (lazy): match minimal
re.search(r"<.*?>", text).group()  # "<div>" (sampai first >)

text2 = "price: $10, $20, $30"
re.findall(r"\$\d+", text2)       # ['$10', '$20', '$30']
re.findall(r"\$\d+?", text2)      # sama karena unambiguous
```

**4. Common re Functions**
```python
# re.search(pattern, string, flags=0)
# Cari pattern pertama, return Match object atau None
match = re.search(r"\d+", "abc123def")
if match:
    print(match.start())  # 3 (index start)
    print(match.end())    # 6 (index end, exclusive)
    print(match.span())   # (3, 6)

# re.match(pattern, string, flags=0)
# Hanya match di awal string
re.match(r"abc", "abc123")  # match
re.match(r"abc", "123abc")  # None

# re.fullmatch(pattern, string, flags=0)
# Seluruh string harus match
re.fullmatch(r"\d+", "123")  # match
re.fullmatch(r"\d+", "123a") # None

# re.findall(pattern, string, flags=0)
# Return list semua matches (string atau tuple jika ada groups)
re.findall(r"\d+", "a1b22c333")  # ['1', '22', '333']
re.findall(r"(\w+)=(\d+)", "x=1 y=2")  # [('x','1'), ('y','2')]

# re.finditer(pattern, string, flags=0)
# Return iterator of Match objects (lebih memory efficient untuk banyak matches)
for match in re.finditer(r"\d+", "a1b22c333"):
    print(match.group(), match.span())

# re.sub(pattern, repl, string, count=0, flags=0)
# Replace matches dengan repl (string atau function)
result = re.sub(r"\d+", "NUM", "a1b22c333")  # "aNUMbNUMcNUM"
result = re.sub(r"(\w+)=(\d+)", r"\2:\1", "x=1 y=2")  # "1:x 2:y"

# Using function as repl
def repl_func(match):
    return f"[{match.group()}]"
result = re.sub(r"\d+", repl_func, "a1b22")  # "a[1]b[22]"

# re.split(pattern, string, maxsplit=0, flags=0)
# Split string by pattern
parts = re.split(r"\s+", "hello   world\tfoo")  # ['hello', 'world', 'foo']
parts = re.split(r"[,;]", "a,b;c,d")  # ['a', 'b', 'c', 'd']
```

**5. Flags (modifiers)**
```python
# re.IGNORECASE (re.I) - case-insensitive
re.search(r"hello", "HELLO", re.I)  # match

# re.MULTILINE (re.M) - ^ dan $ mewakili baris, bukan entire string
text = "line1\nline2"
re.search(r"^line2", text, re.M)  # match (di baris kedua)

# re.DOTALL (re.S) - dot (.) match newline juga
text = "first\nsecond"
re.search(r"first.second", text, re.S)  # match (dot match \n)

# Kombinasi flags
re.search(r"hello", "HELLO", re.I | re.M)
```
