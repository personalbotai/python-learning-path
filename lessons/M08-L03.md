**1. Counter - Hitung frekuensi elemen**
```python
from collections import Counter

# Count items
fruits = ["apple", "banana", "apple", "orange", "banana", "apple"]
fruit_counts = Counter(fruits)
print(fruit_counts)  # Counter({'apple': 3, 'banana': 2, 'orange': 1})

# Access counts
print(fruit_counts["apple"])   # 3
print(fruit_counts["grape"])   # 0 (no KeyError)
print(fruit_counts.most_common(2))  # [('apple', 3), ('banana', 2)]

# Arithmetic operations
c1 = Counter(a=3, b=1)
c2 = Counter(a=1, b=2, c=3)
print(c1 + c2)   # Counter({'a': 4, 'b': 3, 'c': 3})
print(c1 - c2)   # Counter({'a': 2}) (negative counts removed)
print(c1 & c2)   # Intersection: min counts
print(c1 | c2)   # Union: max counts

# Use case: word frequency
text = "hello world hello python world"
word_counts = Counter(text.split())
print(word_counts)  # Counter({'hello': 2, 'world': 2, 'python': 1})
```

**2. defaultdict - Dictionary dengan default value**
```python
from collections import defaultdict

# Default to int (0)
int_dict = defaultdict(int)
int_dict["a"] += 1  # No KeyError, default 0
int_dict["b"] = 5
print(int_dict)  # {'a': 1, 'b': 5}

# Default to list
list_dict = defaultdict(list)
list_dict["fruits"].append("apple")
list_dict["fruits"].append("banana")
list_dict["vegetables"].append("carrot")
print(list_dict)
# {'fruits': ['apple', 'banana'], 'vegetables': ['carrot']}

# Group items by key
pairs = [("a", 1), ("b", 2), ("a", 3), ("b", 4)]
grouped = defaultdict(list)
for key, value in pairs:
    grouped[key].append(value)
print(grouped)  # {'a': [1, 3], 'b': [2, 4]}

# Default to custom function
def default_name():
    return "Unknown"

name_dict = defaultdict(default_name)
print(name_dict["new"])  # "Unknown"
```

**3. deque - Double-ended queue (O(1) operations di kedua ujung)**
```python
from collections import deque

# Create
dq = deque([1, 2, 3, 4, 5])
print(dq)  # deque([1, 2, 3, 4, 5])

# Append/pop di kedua sisi (O(1))
dq.append(6)      # [1,2,3,4,5,6]
dq.appendleft(0)  # [0,1,2,3,4,5,6]
dq.pop()          # 6, dq=[0,1,2,3,4,5]
dq.popleft()      # 0, dq=[1,2,3,4,5]

# Rotate
dq.rotate(2)  # shift ke kanan 2: [4,5,1,2,3]
dq.rotate(-1) # shift ke kiri 1: [5,1,2,3,4]

# Extend
dq.extend([6, 7])      # tambah di kanan
dq.extendleft([0, -1]) # tambah di kiri (reverse order!)

# Clear
dq.clear()

# Use as stack (LIFO) atau queue (FIFO)
stack = deque()
stack.append(1)
stack.append(2)
stack.pop()  # 2

queue = deque()
queue.append("first")
queue.append("second")
queue.popleft()  # "first"
```

**4. OrderedDict (Python 3.7+ dict sudah ordered, tapi OrderedDict masih berguna untuk move_to_end)**
```python
from collections import OrderedDict

od = OrderedDict([('a', 1), ('b', 2), ('c', 3)])
print(od)  # OrderedDict([('a', 1), ('b', 2), ('c', 3)])

# Move key to end (or beginning with last=False)
od.move_to_end('b')  # urutan: a, c, b
print(od.popitem(last=False))  # ('a', 1) - pop first item

# Pop specific key
od.pop('c')  # remove c

# Use case: LRU cache
class LRUCache:
    def __init__(self, capacity: int):
        self.cache = OrderedDict()
        self.capacity = capacity
    
    def get(self, key):
        if key not in self.cache:
            return -1
        self.cache.move_to_end(key)  # mark as recently used
        return self.cache[key]
    
    def put(self, key, value):
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        if len(self.cache) > self.capacity:
            self.cache.popitem(last=False)  # remove least recently used
```

**5. ChainMap - Multiple dictionaries sebagai satu view**
```python
from collections import ChainMap

dict1 = {'a': 1, 'b': 2}
dict2 = {'b': 3, 'c': 4}
chain = ChainMap(dict1, dict2)
print(chain['a'])  # 1 (from dict1)
print(chain['b'])  # 3 (from dict1, first occurrence)
print(chain['c'])  # 4 (from dict2)

# Updates
chain['a'] = 10    # modifies dict1
chain['d'] = 5     # adds to dict1 (first dict)

# New dicts bisa ditambahkan
dict3 = {'b': 5, 'e': 6}
chain = ChainMap(dict1, dict2, dict3)
```

**7. heapq - Priority queue (min-heap)**
```python
import heapq

# Create heap dari list (in-place)
heap = [3, 1, 5, 2, 4]
heapq.heapify(heap)  # O(n)
print(heap)  # [1, 2, 5, 3, 4] (heap property: parent <= children)

# Push & pop
heapq.heappush(heap, 0)
smallest = heapq.heappop(heap)  # 0
print(smallest)

# Peek smallest tanpa pop
smallest = heap[0]  # 1

# N smallest/largest
print(heapq.nsmallest(3, heap))  # [1, 2, 3]
print(heapq.nlargest(2, heap))   # [5, 4]

# Use case: priority queue
tasks = []
heapq.heappush(tasks, (2, "low priority"))  # (priority, task)
heapq.heappush(tasks, (1, "high priority"))
print(heapq.heappop(tasks))  # (1, 'high priority')
```

**8. bisect - Binary search pada sorted list**
```python
import bisect

arr = [1, 3, 4, 6, 8, 9]

# Find insertion point
pos = bisect.bisect_left(arr, 5)  # 3 (index untuk insert 5 agar sorted)
pos = bisect.bisect_right(arr, 4) # 4 (after existing 4)
pos = bisect.bisect(arr, 5)       # same as bisect_right

# Insert while maintaining sorted order
bisect.insort(arr, 5)  # arr = [1,3,4,5,6,8,9]

# Use case: maintain sorted list
scores = []
bisect.insort(scores, 85)
bisect.insort(scores, 70)
bisect.insort(scores, 90)
print(scores)  # [70, 85, 90]
```
