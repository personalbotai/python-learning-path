## Materi

### Konsep Dasar

Python memiliki sistem arguments yang fleksibel. Memahami cara kerjanya penting untuk menulis API fungsi yang robust.

**1. Positional vs Keyword Arguments**
```python
def greet(name, greeting):
    return f"{greeting}, {name}!"

# Positional (urutan penting)
greet("Alice", "Hello")      # "Hello, Alice!"

# Keyword (urutan tidak penting, explicit)
greet(name="Bob", greeting="Hi")  # "Hi, Bob!"
greet(greeting="Good morning", name="Charlie")  # "Good morning, Charlie!"

# Mixed: positional dulu, baru keyword
greet("David", greeting="Hey")  # OK
# greet(name="Eve", "Hi")        # SyntaxError! positional setelah keyword
```

**2. Default Arguments**
```python
def power(base, exponent=2):
    return base ** exponent

power(5)         # 25 (exponent default 2)
power(5, 3)      # 125
power(exponent=3, base=2)  # 8 (keyword)

# Hati-hati: default mutable arguments (BUG COMMON!)
def append_to_list(item, items=[]):  # ❌ BERBAHAYA!
    items.append(item)
    return items

print(append_to_list(1))  # [1]
print(append_to_list(2))  # [1, 2] - items tetap refer ke list yang sama!

# ✅ SOLUSI: gunakan None sebagai default
def append_to_list_fixed(item, items=None):
    if items is None:
        items = []
    items.append(item)
    return items

print(append_to_list_fixed(1))  # [1]
print(append_to_list_fixed(2))  # [2] - setiap call list baru
```

### Detail Teknis

**3. *args (variable positional arguments)**
```python
def sum_all(*args):
    """Accept any number of positional arguments as tuple"""
    return sum(args)

sum_all(1, 2, 3)        # 6
sum_all(1, 2, 3, 4, 5)  # 15
sum_all()               # 0 (empty tuple)

# Kombinasi dengan regular parameters
def greet_all(greeting, *names):
    """greeting positional, names variabel"""
    for name in names:
        print(f"{greeting}, {name}!")

greet_all("Hello", "Alice", "Bob", "Charlie")
# Hello, Alice!
# Hello, Bob!
# Hello, Charlie!

# args adalah tuple, bisa di-iterate, di-index
def print_args(*args):
    print(f"Type: {type(args)}")  # tuple
    print(f"Length: {len(args)}")
    for i, arg in enumerate(args):
        print(f"Arg {i}: {arg}")

print_args(1, "hello", 3.14)
```

**4. **kwargs (variable keyword arguments)**
```python
def print_person(**kwargs):
    """Accept any number of keyword arguments as dict"""
    for key, value in kwargs.items():
        print(f"{key}: {value}")

print_person(name="Alice", age=25, city="Jakarta")
# name: Alice
# age: 25
# city: Jakarta

# kwargs adalah dict
def show_kwargs(**kwargs):
    print(type(kwargs))  # dict
    print(kwargs.get("missing", "default"))

show_kwargs(a=1, b=2)
```

**5. Kombinasi Semua Jenis Parameters (urutan penting!)**

```python
def complex_function(
    pos1,           # 1. positional-only (Python 3.8+)
    pos2,           # 2. positional-or-keyword
    /,              # separator: sebelum ini positional-only
    kwd1,           # 3. keyword-or-positional
    kwd2="default", # 4. keyword with default
    *,              # separator: setelah ini keyword-only
    kwonly1,        # 5. keyword-only (required)
    kwonly2="val",  # 6. keyword-only with default
    **kwargs        # 7. variabel keyword
):
    print(f"pos1={pos1}, pos2={pos2}")
    print(f"kwd1={kwd1}, kwd2={kwd2}")
    print(f"kwonly1={kwonly1}, kwonly2={kwonly2}")
    print(f"kwargs={kwargs}")

# Contoh calls:
complex_function(1, 2, 3, kwd1=4, kwonly1=5)  # OK
complex_function(1, 2, 3, 4, kwonly1=5)       # OK (kwd1 positional)
# complex_function(1, 2, kwd1=3, kwonly1=4)   # Error! pos2 missing
```

**Positional-Only Parameters (Python 3.8+):**
```python
# Gunakan / untuk menandai parameters yang hanya bisa positional
def add(a, b, /):
    return a + b

add(5, 3)      # OK
add(a=5, b=3)  # TypeError! a dan b positional-only

# Use case: API yang ingin mencegah user menggunakan keyword agar lebih clean
def pow(x, y, /):
    return x ** y
pow(2, 3)  # 8
# pow(x=2, y=3)  # Error
```

**Keyword-Only Parameters:**
```python
# Gunakan * untuk menandai parameters setelahnya hanya bisa keyword
def greet(name, *, greeting="Hello", punctuation="!"):
    return f"{greeting}, {name}{punctuation}"

greet("Alice")                    # "Hello, Alice!"
greet("Bob", greeting="Hi")       # "Hi, Bob!"
greet("Charlie", greeting="Hey", punctuation="?")  # "Hey, Charlie?"
# greet("David", "Hi")            # Error! "Hi" harus keyword
```

**6. Argument Unpacking saat Function Call**
```python
def add(a, b, c):
    return a + b + c

# Unpack list/tuple
values = [1, 2, 3]
result = add(*values)  # unpack: add(1, 2, 3)

# Unpack dict (jika key cocok dengan parameter names)
params = {"a": 10, "b": 20, "c": 30}
result = add(**params)  # unpack: add(a=10, b=20, c=30)

# Kombinasi
def func(a, b, c, d):
    print(a, b, c, d)

args = [1, 2]
kwargs = {"c": 3, "d": 4}
func(*args, **kwargs)  # 1 2 3 4
```

**7. Function Annotations (type hints untuk parameters & return)**
```python
def add(a: int, b: int) -> int:
    """Type hints - tidak enforce, hanya dokumentasi"""
    return a + b

# Access annotations
print(add.__annotations__)  # {'a': <class 'int'>, 'b': <class 'int'>, 'return': <class 'int'>}

def greet(name: str, age: int = 18) -> str:
    return f"{name} is {age} years old"

# Bisa dengan complex types
from typing import List, Dict, Optional
def process_items(items: List[str], config: Optional[Dict] = None) -> bool:
    if config is None:
        config = {}
    return True
```

@repeat(times=3)
def greet(name):
    return f"Hello, {name}!"

@repeat(times=2)
def add(a, b):
