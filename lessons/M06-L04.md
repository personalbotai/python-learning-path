## Materi

### Konsep Dasar

Python memiliki module `zipfile` untuk operasi ZIP, dan `shutil` untuk format archive lainnya (tar, gztar, dll). ZIP adalah format compression yang umum untuk bundling multiple files.

**ZIP vs Tar:**
- ZIP: compression + archiving, widely supported, built-in module `zipfile`
- Tar: archiving only (no compression), biasanya dikombinasi dengan gzip/bzip2/xz

### Detail Teknis

**1. zipfile Module - Basic Operations**

```python
import zipfile
from pathlib import Path

# Create new ZIP file
with zipfile.ZipFile("archive.zip", "w", compression=zipfile.ZIP_DEFLATED) as zf:
    zf.write("file1.txt")  # add file
    zf.write("file2.txt")
    # write juga dari memory
    zf.writestr("notes.txt", "This is a note inside zip")

# Read ZIP file
with zipfile.ZipFile("archive.zip", "r") as zf:
    # List all files in archive
    print(zf.namelist())  # ['file1.txt', 'file2.txt', 'notes.txt']
    
    # Get info about specific file
    info = zf.getinfo("file1.txt")
    print(info.file_size)      # original size
    print(info.compress_size)  # compressed size
    print(info.compress_type)  # zipfile.ZIP_DEFLATED
    
    # Extract all files
    zf.extractall("extracted_folder")
    
    # Extract single file
    zf.extract("file1.txt", path=".")
    
    # Read file content dari dalam zip (tanpa extract ke disk)
    content = zf.read("notes.txt")
    print(content.decode("utf-8"))  # bytes -> string

# Append to existing ZIP
with zipfile.ZipFile("archive.zip", "a") as zf:
    zf.write("new_file.txt")
```

**2. Compression Methods**
```python
# ZIP_DEFLATED: DEFLATE compression (standard, good compression)
# ZIP_STORED: no compression (just store)
# ZIP_BZIP2: BZIP2 compression (better but slower) - Python 3.7+
# ZIP_LZMA: LZMA compression (best but slow) - Python 3.7+

with zipfile.ZipFile("archive.zip", "w", compression=zipfile.ZIP_DEFLATED) as zf:
    zf.write("large_file.txt", compress_type=zipfile.ZIP_DEFLATED)

# Cek compression ratio
with zipfile.ZipFile("archive.zip") as zf:
    for info in zf.infolist():
        ratio = info.compress_size / info.file_size * 100 if info.file_size > 0 else 0
        print(f"{info.filename}: {info.file_size} -> {info.compress_size} ({ratio:.1f}%)")
```

**3. Password Protection (AES Encryption)**
```python
# Standard zipfile module tidak support encryption
# Gunakan `pyminizip` atau `pyzipper` untuk password protection

# Contoh dengan pyzipper (pip install pyzipper)
import pyzipper

with pyzipper.AESZipFile("secure.zip", "w", compression=pyzipper.ZIP_LZMA, encryption=pyzipper.WZ_AES) as zf:
    zf.setpassword(b"secret123")
    zf.write("confidential.txt")

with pyzipper.AESZipFile("secure.zip") as zf:
    zf.setpassword(b"secret123")
    content = zf.read("confidential.txt")
```

**4. Tar Archives (shutil)**
```python
import shutil
from datetime import datetime

# Create tar archive (uncompressed)
shutil.make_archive("backup", "tar", "folder_to_archive")

# Create compressed tar archives
shutil.make_archive("backup", "gztar", "folder")   # .tar.gz
shutil.make_archive("backup", "bztar", "folder")   # .tar.bz2
shutil.make_archive("backup", "xztar", "folder")   # .tar.xz (best compression)

# Extract any archive format (auto-detects format)
shutil.unpack_archive("backup.tar.gz", "extract_dest")
```

**5. Advanced Zipfile - Custom Filtering**
```python
def create_selective_zip(source_dir, zip_path, extensions=None):
    """
    Buat ZIP hanya dari file dengan ekstensi tertentu.
    extensions: ['.txt', '.py', '.md'] atau None untuk semua
    """
    with zipfile.ZipFile(zip_path, "w", compression=zipfile.ZIP_DEFLATED) as zf:
        source = Path(source_dir)
        for file_path in source.rglob("*"):
            if file_path.is_file():
                if extensions is None or file_path.suffix in extensions:
                    # arcname = path di dalam zip (preserve relative structure)
                    arcname = file_path.relative_to(source)
                    zf.write(file_path, arcname)
```

**6. Zipfile for In-Memory Archives**
```python
import io
import zipfile

# Buat ZIP di memory (bytes)
buffer = io.BytesIO()
with zipfile.ZipFile(buffer, "w") as zf:
    zf.writestr("hello.txt", "Hello World")
    zf.writestr("data.json", '{"key": "value"}')

# Get bytes
zip_bytes = buffer.getvalue()
print(f"ZIP size: {len(zip_bytes)} bytes")

# Baca dari bytes
buffer.seek(0)
with zipfile.ZipFile(buffer) as zf:
    print(zf.namelist())
```

### Soal 2: Incremental Backup (modified files only)

Buat script `incremental_backup.py` yang membuat ZIP hanya dari file yang dimodifikasi sejak backup terakhir.

```python
import zipfile
import json
from pathlib import Path
from datetime import datetime

def incremental_backup(source_dir, zip_path, manifest_path="backup_manifest.json"):
    """
    Buat backup hanya file yang berubah sejak backup terakhir.
    Simpan metadata (mtime) di manifest.json untuk track perubahan.
    """
    source = Path(source_dir)
    manifest = {}
    
    # Load existing manifest jika ada
    if Path(manifest_path).exists():
        with open(manifest_path, "r") as f:
            manifest = json.load(f)  # dict: relative_path -> mtime
    
    # Cek file mana yang berubah
    changed_files = []
    for file_path in source.rglob("*"):
        if file_path.is_file():
            rel_path = str(file_path.relative_to(source))
            current_mtime = file_path.stat().st_mtime
            
            # Jika file belum ada di manifest atau mtime berbeda -> changed
            if rel_path not in manifest or manifest[rel_path] != current_mtime:
                changed_files.append((file_path, rel_path, current_mtime))
    
    if not changed_files:
        print("No changes detected. Backup skipped.")
        return
    
    # Create ZIP dengan changed files saja
    with zipfile.ZipFile(zip_path, "a", compression=zipfile.ZIP_DEFLATED) as zf:
        for file_path, rel_path, mtime in changed_files:
            zf.write(file_path, rel_path)
            manifest[rel_path] = mtime
    
    # Update manifest
    with open(manifest_path, "w") as f:
        json.dump(manifest, f, indent=2)
    
    print(f"Backup created: {len(changed_files)} files updated.")

def main():
    incremental_backup("myproject", "backup.zip")
```
