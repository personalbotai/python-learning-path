## Materi

### Konsep Dasar

**Inheritance** memungkinkan class baru (child/subclass) mewarisi attributes dan methods dari class lain (parent/superclass). Ini memungkinkan code reuse dan hierarchical relationships.

**Polymorphism** memungkinkan objects dari berbagai class bisa diperlakukan secara seragam melalui common interface (misal: semua subclass mengimplementasikan method yang sama).

**Contoh sederhana:**
```python
class Animal:
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        raise NotImplementedError("Subclasses must implement")

class Dog(Animal):
    def speak(self):
        return "Woof!"

class Cat(Animal):
    def speak(self):
        return "Meow!"

# Polymorphism: berbagai class, interface yang sama
animals = [Dog("Buddy"), Cat("Whiskers")]
for animal in animals:
    print(f"{animal.name} says: {animal.speak()}")
# Buddy says: Woof!
# Whiskers says: Meow!
```

### Detail Teknis

**1. Basic Inheritance**

```python
class Parent:
    def __init__(self, name):
        self.name = name
    
    def greet(self):
        return f"Hello from {self.name}"

class Child(Parent):  # Child extends Parent
    def __init__(self, name, age):
        super().__init__(name)  # Call parent __init__
        self.age = age
    
    # Override parent method
    def greet(self):
        parent_greeting = super().greet()
        return f"{parent_greeting} (I'm {self.age} years old)"

child = Child("Alice", 10)
print(child.greet())  # "Hello from Alice (I'm 10 years old)"
```

**2. Method Resolution Order (MRO)**

```python
class A:
    def method(self):
        return "A.method"

class B(A):
    def method(self):
        return "B.method"

class C(A):
    def method(self):
        return "C.method"

class D(B, C):  # Multiple inheritance
    pass

d = D()
print(d.method())  # "B.method" - MRO: D -> B -> C -> A
print(D.__mro__)   # (<class 'D'>, <class 'B'>, <class 'C'>, <class 'A'>, <class 'object'>)
```

**3. Abstract Base Classes (ABC)**

```python
from abc import ABC, abstractmethod

class Shape(ABC):
    """Abstract base class - cannot be instantiated"""
    
    @abstractmethod
    def area(self):
        """Calculate area - must be implemented by subclasses"""
        pass
    
    @abstractmethod
    def perimeter(self):
        """Calculate perimeter"""
        pass
    
    def describe(self):
        return f"{self.__class__.__name__} with area {self.area():.2f}"

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    
    def area(self):
        import math
        return math.pi * self.radius ** 2
    
    def perimeter(self):
        import math
        return 2 * math.pi * self.radius

# shape = Shape()  # TypeError: Can't instantiate abstract class
rect = Rectangle(4, 5)
print(rect.area())      # 20
print(rect.describe())  # Rectangle with area 20.00
```

**4. `isinstance()` and `issubclass()`**

```python
class Animal:
    pass

class Mammal(Animal):
    pass

class Dog(Mammal):
    pass

dog = Dog()

print(isinstance(dog, Dog))      # True
print(isinstance(dog, Mammal))   # True (inheritance)
print(isinstance(dog, Animal))   # True
print(isinstance(dog, str))      # False

print(issubclass(Dog, Mammal))   # True
print(issubclass(Mammal, Animal)) # True
print(issubclass(Dog, Animal))   # True
```

**5. Multiple Inheritance & Diamond Problem**

```python
class A:
    def method(self):
        return "A.method"

class B(A):
    def method(self):
        return "B.method"

class C(A):
    def method(self):
        return "C.method"

class D(B, C):  # Diamond: D -> B -> A, D -> C -> A
    pass

d = D()
print(d.method())  # "B.method" (MRO: D, B, C, A, object)
print(D.mro())     # [D, B, C, A, object]

# Using super() in multiple inheritance
class Base:
    def __init__(self):
        print("Base.__init__")
        self.base_attr = "base"

class Left(Base):
    def __init__(self):
        super().__init__()
        print("Left.__init__")
        self.left_attr = "left"

class Right(Base):
    def __init__(self):
        super().__init__()
        print("Right.__init__")
        self.right_attr = "right"

class Child(Left, Right):
    def __init__(self):
        super().__init__()  # Calls Left.__init__, which calls Right.__init__ via super() MRO
        print("Child.__init__")

c = Child()
# Output:
# Base.__init__
# Right.__init__
# Left.__init__
# Child.__init__
# Note: Base.__init__ called once due to MRO (diamond problem solved by C3 linearization)
```

**6. Mixins (Multiple Inheritance for Code Reuse)**

```python
# Mixin: small class that provides additional functionality
class JSONSerializableMixin:
    import json
    def to_json(self):
        return self.json.dumps(self.__dict__)

class LoggableMixin:
    def log(self, message):
        print(f"[{self.__class__.__name__}] {message}")

class User(JSONSerializableMixin, LoggableMixin):
    def __init__(self, name, email):
        self.name = name
        self.email = email

user = User("Alice", "alice@example.com")
print(user.to_json())  # {"name": "Alice", "email": "alice@example.com"}
user.log("User created")  # [User] User created
```

**7. Polymorphism & Duck Typing**

```python
# Duck typing: "If it walks like a duck and quacks like a duck, it's a duck"
class Rectangle:
    def area(self):
        return self.width * self.height

class Circle:
    def area(self):
        import math
        return math.pi * self.radius ** 2

def print_area(shape):
    # Doesn't care about type, only that shape has .area() method
    print(f"Area: {shape.area()}")

rect = Rectangle()
rect.width, rect.height = 4, 5
circle = Circle()
circle.radius = 3

print_area(rect)   # Works
print_area(circle) # Works - polymorphic behavior
```
