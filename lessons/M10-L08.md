**2. Docker Fundamentals**

```dockerfile
# Dockerfile
# Stage 1: Build
FROM python:3.11-slim as builder

WORKDIR /app
COPY pyproject.toml .
COPY src/ src/

RUN pip install --user --upgrade pip && \
    pip install --user build && \
    python -m build --wheel --out-dir /app/wheelhouse

# Stage 2: Runtime
FROM python:3.11-slim

WORKDIR /app

# Copy built wheel dari builder stage
COPY --from=builder /app/wheelhouse/*.whl .

# Install aplikasi
RUN pip install --no-cache-dir *.whl

# Copy source (jika perlu)
COPY src/ src/

# Expose port
EXPOSE 8000

# Run command
CMD ["python", "-m", "mypackage.api"]
```

Build dan run:
```bash
docker build -t myapp:latest .
docker run -p 8000:8000 myapp:latest
```

**3. Docker Compose (multi-service)**

```yaml
# docker-compose.yml
version: '3.8'
services:
  web:
    build: .
    ports:
      - "8000:8000"
    environment:
      - DATABASE_URL=postgresql://user:pass@db:5432/mydb
      - REDIS_URL=redis://redis:6379
    depends_on:
      - db
      - redis
    volumes:
      - ./logs:/app/logs
    restart: unless-stopped

  db:
    image: postgres:15
    environment:
      POSTGRES_USER: user
      POSTGRES_PASSWORD: pass
      POSTGRES_DB: mydb
    volumes:
      - postgres_data:/var/lib/postgresql/data
    restart: unless-stopped

  redis:
    image: redis:7-alpine
    restart: unless-stopped

volumes:
  postgres_data:
```

Run:
```bash
docker-compose up -d  # detached
docker-compose logs -f web  # follow logs
docker-compose down  # stop
```

**4. CI/CD dengan GitHub Actions**

```yaml
# .github/workflows/ci.yml
name: CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        python-version: ["3.9", "3.10", "3.11"]

    steps:
      - uses: actions/checkout@v3
      - name: Set up Python ${{ matrix.python-version }}
        uses: actions/setup-python@v4
        with:
          python-version: ${{ matrix.python-version }}
      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -e .[dev]
      - name: Run tests
        run: pytest -v --cov=src --cov-report=xml
      - name: Upload coverage to Codecov
        uses: codecov/codecov-action@v3
        with:
          token: ${{ secrets.CODECOV_TOKEN }}

  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: "3.11"
      - name: Install linters
        run: |
          pip install black isort flake8 mypy
      - name: Run black (check only)
        run: black --check src tests
      - name: Run isort (check only)
        run: isort --check-only src tests
      - name: Run flake8
        run: flake8 src tests
      - name: Run mypy
        run: mypy src

  build:
    needs: [test, lint]
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v3
      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v2
      - name: Login to DockerHub
        uses: docker/login-action@v2
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_PASSWORD }}
      - name: Build and push
        uses: docker/build-push-action@v4
        with:
          context: .
          push: true
          tags: username/myapp:latest, username/myapp:${{ github.sha }}

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - name: Deploy to server (example)
        run: |
          echo "Deploying to production..."
          # ssh user@server "docker pull username/myapp:latest && docker-compose up -d"
```

**7. Process Managers (Production)**

```bash
# Gunicorn (WSGI server for Flask/FastAPI with workers)
gunicorn main:app --workers 4 --bind 0.0.0.0:8000 --access-logfile -

# Uvicorn (ASGI server for FastAPI)
uvicorn main:app --host 0.0.0.0 --port 8000 --workers 4

# Systemd service (Linux)
# /etc/systemd/system/myapp.service
[Unit]
Description=My Python App
After=network.target

[Service]
User=appuser
WorkingDirectory=/opt/myapp
Environment="PATH=/opt/myapp/venv/bin"
EnvironmentFile=/opt/myapp/.env
ExecStart=/opt/myapp/venv/bin/gunicorn main:app --workers 4
Restart=always

[Install]
WantedBy=multi-user.target

# sudo systemctl enable myapp
# sudo systemctl start myapp
```

**10. Security Hardening**

```dockerfile
# Multi-stage build untuk smaller image
FROM python:3.11-slim as builder
# ... build stage

FROM python:3.11-slim
# Create non-root user
RUN adduser --disabled-password --gecos '' appuser
USER appuser
WORKDIR /app
COPY --from=builder /app/wheelhouse/*.whl .
RUN pip install --no-cache-dir *.whl
# Non-root user increase security
```


### Soal 3: Production Deployment Script

Buat script `deploy.sh` untuk deployment manual ke server:

```bash
#!/bin/bash
set -e  # exit on error

APP_NAME="myapp"
SERVER="user@myserver.com"
DEPLOY_DIR="/opt/$APP_NAME"

echo "Building Docker image..."
docker build -t $APP_NAME:latest .

echo "Saving image to tar..."
docker save $APP_NAME:latest | gzip > image.tar.gz

echo "Transferring to server..."
scp image.tar.gz $SERVER:/tmp/

echo "Deploying on server..."
ssh $SERVER bash << 'EOF'
    set -e
    cd /opt/myapp
    docker load < /tmp/image.tar.gz
    docker-compose down
    docker-compose up -d
    docker image prune -f  # remove old images
    echo "Deployment complete!"
EOF

echo "Cleaning up local..."
rm image.tar.gz

echo "Done!"
```

Buat juga `rollback.sh` untuk rollback ke versi sebelumnya (simpan 2 image terbaru).
