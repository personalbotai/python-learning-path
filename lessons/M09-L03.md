## Materi

### Konsep Dasar

**Asyncio** adalah library untuk writing concurrent code using `async`/`await` syntax. Ini adalah **single-threaded** concurrency (cooperative multitasking). GIL tetap ada, tapi tasks voluntarily yield control saat menunggu I/O.

**Key differences:**
- **Threading**: OS context switching, preemptive, threads share memory (need locks)
- **Asyncio**: Single thread, cooperative, tasks yield with `await`, no locks needed (single-threaded)

**Use asyncio when:**
- High concurrency I/O-bound (thousands of simultaneous connections)
- Network servers, web scrapers, APIs
- Want to avoid thread overhead

**Use threading when:**
- Blocking libraries that don't support async
- Need true parallelism for CPU-bound (but use multiprocessing instead)

### Detail Teknis

**1. Basic Coroutine**

```python
import asyncio

async def hello():
    print("Hello")
    await asyncio.sleep(1)  # simulate non-blocking I/O
    print("World")

# Running coroutine
asyncio.run(hello())  # Python 3.7+
# Output after 1s:
# Hello
# World
```

**2. `await` - Yield control to event loop**

```python
async def task(name, delay):
    print(f"{name} started")
    await asyncio.sleep(delay)  # non-blocking sleep
    print(f"{name} done after {delay}s")
    return f"{name} result"

async def main():
    # Sequential (bisa diganti concurrent)
    await task("A", 2)
    await task("B", 1)  # menunggu A selesai dulu

asyncio.run(main())
```

**3. Running Multiple Coroutines Concurrently**

```python
async def main():
    # create tasks (schedule untuk run concurrent)
    task_a = asyncio.create_task(task("A", 2))
    task_b = asyncio.create_task(task("B", 1))
    
    # Wait for all tasks
    await task_a
    await task_b
    
    # atau gunakan gather
    results = await asyncio.gather(
        task("A", 2),
        task("B", 1),
        task("C", 3)
    )
    print(results)  # ['A result', 'B result', 'C result']

asyncio.run(main())
# Output (concurrent):
# A started
# B started
# B done after 1s
# A done after 2s
# C done after 3s
```

**4. `asyncio.sleep()` vs `time.sleep()`**

```python
async def correct():
    await asyncio.sleep(1)  # yields to event loop, other tasks bisa run

async def wrong():
    time.sleep(1)  # BLOCKING! entire event loop stops for 1s
```

**5. Async Context Managers & Iterators**

```python
class AsyncCounter:
    def __init__(self, limit):
        self.limit = limit
        self.current = 0
    
    def __aiter__(self):
        return self
    
    async def __anext__(self):
        if self.current >= self.limit:
            raise StopAsyncIteration
        await asyncio.sleep(0.1)  # simulate async operation
        self.current += 1
        return self.current

async def use_async_iter():
    async for value in AsyncCounter(5):
        print(value)  # 1,2,3,4,5 (dengan delay 0.1s)

asyncio.run(use_async_iter())

# Async context manager
class AsyncDatabase:
    async def __aenter__(self):
        await asyncio.sleep(0.1)  # connect
        print("Connected")
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await asyncio.sleep(0.1)  # disconnect
        print("Disconnected")
    
    async def query(self, sql):
        await asyncio.sleep(0.5)
        return f"Result for: {sql}"

async def use_db():
    async with AsyncDatabase() as db:
        result = await db.query("SELECT * FROM users")
        print(result)

asyncio.run(use_db())
```

**6. `asyncio.wait()` & Timeouts**

```python
async def main():
    tasks = [
        asyncio.create_task(asyncio.sleep(5, result="5s")),
        asyncio.create_task(asyncio.sleep(2, result="2s")),
        asyncio.create_task(asyncio.sleep(1, result="1s"))
    ]
    
    # Wait for all tasks with timeout
    done, pending = await asyncio.wait(tasks, timeout=2)
    # done: tasks yang complete sebelum timeout
    # pending: tasks yang masih running
    
    for task in done:
        print(task.result())  # 2s, 1s (5s masih pending)
    
    # Cancel pending tasks
    for task in pending:
        task.cancel()

asyncio.run(main())
```

**7. Semaphore untuk Rate Limiting**

```python
async def fetch(url, semaphore):
    async with semaphore:  # limit concurrent requests
        print(f"Fetching {url}")
        await asyncio.sleep(1)
        return f"Content from {url}"

async def main():
    semaphore = asyncio.Semaphore(3)  # max 3 concurrent
    urls = [f"url{i}" for i in range(10)]
    
    tasks = [fetch(url, semaphore) for url in urls]
    results = await asyncio.gather(*tasks)
    print(results)

asyncio.run(main())
```

**8. Async Queue (Producer-Consumer)**

```python
import asyncio

async def producer(queue, n):
    for i in range(n):
        await queue.put(i)
        print(f"Produced: {i}")
        await asyncio.sleep(0.1)
    await queue.put(None)  # sentinel

async def consumer(queue):
    while True:
        item = await queue.get()
        if item is None:
            break
        print(f"Consumed: {item}")
        queue.task_done()

async def main():
    queue = asyncio.Queue(maxsize=3)
    prod = asyncio.create_task(producer(queue, 10))
    cons = asyncio.create_task(consumer(queue))
    
    await prod
    await cons

asyncio.run(main())
```

**9. Running Sync Code in Async (ThreadPoolExecutor)**

```python
import time
import asyncio
from concurrent.futures import ThreadPoolExecutor

def blocking_io():
    time.sleep(2)  # blocking operation
    return "Result"

async def main():
    loop = asyncio.get_running_loop()
    # Run blocking function in thread pool
    result = await loop.run_in_executor(None, blocking_io)
    print(result)  # "Result" after 2s

asyncio.run(main())
```

**10. Async HTTP with aiohttp (conceptual)**

```python
# pip install aiohttp
import aiohttp
import asyncio

async def fetch_url(session, url):
    async with session.get(url) as response:
        return await response.text()

async def main():
    urls = ["http://example.com", "http://example.org"]
    
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_url(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
        for url, content in zip(urls, results):
            print(f"{url}: {len(content)} bytes")
```

### Soal 2: Async Producer-Consumer Pipeline

Implement pipeline dengan producer, processor (multiple workers), dan consumer:

```python
import asyncio
import random

async def producer(queue: asyncio.Queue, n: int):
    """Produce n items"""
    for i in range(n):
        item = random.randint(1, 100)
        await queue.put(item)
        print(f"Produced: {item}")
        await asyncio.sleep(random.random())
    await queue.put(None)  # sentinel

async def processor(queue: asyncio.Queue, result_queue: asyncio.Queue, worker_id: int):
    """Consume from queue, process, put to result_queue"""
    while True:
        item = await queue.get()
        if item is None:
            await result_queue.put(None)  # propagate sentinel
            break
        # Process: multiply by 2 after delay
        await asyncio.sleep(0.5)
        result = item * 2
        print(f"Worker {worker_id}: processed {item} -> {result}")
        await result_queue.put(result)
        queue.task_done()

async def consumer(result_queue: asyncio.Queue, expected: int):
    """Consume results until None received expected times"""
    received = 0
    while received < expected:
        result = await result_queue.get()
        if result is None:
            break
        print(f"Consumed: {result}")
        received += 1
        result_queue.task_done()

async def main():
    queue = asyncio.Queue()
    result_queue = asyncio.Queue()
    n_items = 10
    n_workers = 3
    
    # Start producer
    prod = asyncio.create_task(producer(queue, n_items))
    
    # Start processors
    processors = [
        asyncio.create_task(processor(queue, result_queue, i))
        for i in range(n_workers)
    ]
    
    # Start consumer (menunggu n_items results)
    cons = asyncio.create_task(consumer(result_queue, n_items))
    
    # Wait for producer
    await prod
    # Wait for all processors to finish processing
    await queue.join()  # semua item diproses
    # Send sentinel to result_queue via processors (done inside processor)
    # Wait for processors
    await asyncio.gather(*processors)
    # Wait for consumer
    await cons

asyncio.run(main())
```
