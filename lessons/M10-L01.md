## Materi

### Konsep Dasar

**pytest** adalah testing framework populer untuk Python. Lebih powerful dibanding `unittest` (built-in) dengan sintaks yang lebih simple dan flexible.

**Instalasi:**
```bash
pip install pytest
```

**Struktur project:**
```
myproject/
├── src/
│   └── mymodule.py
└── tests/
    ├── __init__.py
    ├── test_basic.py
    └── test_advanced.py
```

**Running tests:**
```bash
pytest                    # semua test di tests/ atau tests*.py
pytest tests/             # specific directory
pytest tests/test_basic.py  # specific file
pytest -v                 # verbose
pytest -k "test_add"      # run tests matching pattern
```

### Detail Teknis

**1. Basic Test Functions**

```python
# test_math.py
def add(a, b):
    return a + b

def test_add():
    assert add(2, 3) == 5
    assert add(-1, 1) == 0
    assert add(0, 0) == 0

# Run: pytest test_math.py -v
```

**2. Assertions (lebih dari built-in assert)**

```python
def test_assertions():
    # Basic equality
    assert 1 + 1 == 2
    
    # Approximate equality untuk float
    assert 0.1 + 0.2 == pytest.approx(0.3)
    
    # Containment
    assert "hello" in "hello world"
    assert 3 in [1, 2, 3]
    
    # Boolean
    assert True
    assert not False
    
    # Type checking
    assert isinstance("text", str)
    assert isinstance(42, int)
```

**3. Fixtures (setup/teardown)**

```python
import pytest

@pytest.fixture
def sample_data():
    """Fixture yang di-run sebelum test, return value ke test function"""
    data = {"users": ["Alice", "Bob"], "count": 2}
    return data

@pytest.fixture(scope="module")
def db_connection():
    """Scope: function (default), class, module, session"""
    conn = create_connection()  # simulate
    yield conn  # provide value
    conn.close()  # cleanup setelah test selesai

def test_user_count(sample_data):
    assert sample_data["count"] == 2

def test_user_list(sample_data):
    assert "Alice" in sample_data["users"]
```

**4. Parametrize (data-driven tests)**

```python
import pytest

@pytest.mark.parametrize("a,b,expected", [
    (1, 2, 3),
    (0, 0, 0),
    (-1, 1, 0),
    (100, -50, 50)
])
def test_add_parametrized(a, b, expected):
    assert add(a, b) == expected

# Multiple parameters
@pytest.mark.parametrize("input_str,expected_len", [
    ("hello", 5),
    ("", 0),
    ("a b c", 5),  # including spaces
])
def test_string_length(input_str, expected_len):
    assert len(input_str) == expected_len
```

**5. Testing Exceptions**

```python
def divide(a, b):
    if b == 0:
        raise ValueError("Cannot divide by zero")
    return a / b

def test_divide_by_zero():
    with pytest.raises(ValueError, match="Cannot divide by zero"):
        divide(10, 0)
    
    # Match pattern (regex)
    with pytest.raises(ValueError, match="divide by zero"):
        divide(10, 0)

def test_divide_normal():
    assert divide(10, 2) == 5.0
```

**6. Fixtures with Parameters**

```python
@pytest.fixture(params=[1, 2, 3])
def multiplier(request):
    return request.param

def test_with_multiplier(multiplier):
    assert multiplier * 2 > 0
# Test akan di-run 3 kali dengan multiplier=1,2,3
```

**7. Monkeypatching & Mocking**

```python
from unittest.mock import MagicMock, patch

def get_user_from_api(user_id):
    import requests
    resp = requests.get(f"https://api.example.com/users/{user_id}")
    return resp.json()["name"]

def test_get_user_from_api(monkeypatch):
    # Mock requests.get
    mock_get = MagicMock()
    mock_get.return_value.json.return_value = {"name": "Alice"}
    monkeypatch.setattr("requests.get", mock_get)
    
    result = get_user_from_api(123)
    assert result == "Alice"
    mock_get.assert_called_once_with("https://api.example.com/users/123")

# Atau dengan patch
@patch("requests.get")
def test_get_user_from_api_decorator(mock_get):
    mock_get.return_value.json.return_value = {"name": "Bob"}
    result = get_user_from_api(456)
    assert result == "Bob"
```

**8. Markers (skip, xfail, parametrize)**

```python
import pytest
import sys

@pytest.mark.skip(reason="Not implemented yet")
def test_future_feature():
    assert False

@pytest.mark.skipif(sys.platform == "win32", reason="Does not run on Windows")
def test_linux_only():
    assert True

@pytest.mark.xfail(reason="Known bug, will fix later")
def test_expected_to_fail():
    assert 1 + 1 == 3  # fail tapi tidak dianggap error
```

**9. Temporary Directories & Files**

```python
def test_file_operations(tmp_path):
    # tmp_path: pytest fixture untuk temporary directory
    file = tmp_path / "test.txt"
    file.write_text("Hello")
    content = file.read_text()
    assert content == "Hello"
    # directory otomatis dihapus setelah test

def test_tmpdir_fixture(tmpdir):
    # older style
    file = tmpdir.join("test.txt")
    file.write("Hello")
    assert file.read() == "Hello"
```

```python
    assert captured.err == ""

def test_capfd(capfd):

```

### Soal 2: Fixture untuk Database Mock

Buat fixture yang provides mock database connection:

```python
import pytest
from unittest.mock import MagicMock

class UserRepository:
    def __init__(self, db):
        self.db = db
    
    def get_user(self, user_id):
        cursor = self.db.execute("SELECT * FROM users WHERE id = ?", (user_id,))
        return cursor.fetchone()
    
    def create_user(self, name, email):
        self.db.execute(
            "INSERT INTO users (name, email) VALUES (?, ?)",
            (name, email)
        )
        self.db.commit()

@pytest.fixture
def mock_db():
    """Fixture: provides mock database connection"""
    db = MagicMock()
    cursor = MagicMock()
    db.cursor.return_value = cursor
    cursor.fetchone.return_value = {"id": 1, "name": "Alice", "email": "alice@example.com"}
    return db

@pytest.fixture
def user_repo(mock_db):
    """Fixture: repository dengan mock db"""
    return UserRepository(mock_db)

def test_get_user(user_repo):
    user = user_repo.get_user(1)
    assert user["name"] == "Alice"
    user_repo.db.execute.assert_called_once_with(
        "SELECT * FROM users WHERE id = ?", (1,)
    )

def test_create_user(user_repo):
    user_repo.create_user("Bob", "bob@example.com")
    user_repo.db.execute.assert_called_once()
    user_repo.db.commit.assert_called_once()
```
