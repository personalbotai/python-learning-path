## Materi

### Konsep Dasar

**REST API** adalah arsitektur untuk web services yang menggunakan HTTP methods secara standar. Python memiliki framework populer untuk membuat API:

- **FastAPI**: modern, fast, auto-documentation, type hints, async support (Python 3.7+)
- **Flask**: lightweight, flexible, extensive ecosystem (older but mature)

### Detail Teknis

**1. FastAPI Basics**

```python
# pip install fastapi uvicorn[standard]

from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import Optional

app = FastAPI(title="My API", version="1.0.0")

# Pydantic models
class Item(BaseModel):
    name: str
    description: Optional[str] = None
    price: float
    tax: Optional[float] = None

class User(BaseModel):
    id: int
    username: str
    email: str

# In-memory "database"
items_db = {}
users_db = {}

# GET endpoint
@app.get("/")
def read_root():
    return {"message": "Welcome to API", "docs": "/docs"}

@app.get("/items/{item_id}")
def read_item(item_id: int):
    if item_id not in items_db:
        raise HTTPException(status_code=404, detail="Item not found")
    return items_db[item_id]

# POST endpoint
@app.post("/items/")
def create_item(item: Item):
    item_id = len(items_db) + 1
    items_db[item_id] = item.dict()
    return {"id": item_id, **item.dict()}

# PUT endpoint
@app.put("/items/{item_id}")
def update_item(item_id: int, item: Item):
    if item_id not in items_db:
        raise HTTPException(status_code=404, detail="Item not found")
    items_db[item_id] = item.dict()
    return {"id": item_id, **item.dict()}

# DELETE endpoint
@app.delete("/items/{item_id}")
def delete_item(item_id: int):
    if item_id not in items_db:
        raise HTTPException(status_code=404, detail="Item not found")
    del items_db[item_id]
    return {"message": "Item deleted"}

# Run: uvicorn main:app --reload
```

**2. Query Parameters & Path Parameters**

```python
from fastapi import FastAPI, Query

app = FastAPI()

# Path parameter
@app.get("/users/{user_id}")
def get_user(user_id: int):
    return {"user_id": user_id}

# Query parameter (optional dengan default)
@app.get("/search/")
def search_items(q: Optional[str] = None, limit: int = 10):
    results = [item for item in items_db.values() if q is None or q in item["name"]]
    return {"results": results[:limit]}

# Multiple query parameters with validation
@app.get("/filter/")
def filter_items(
    min_price: float = Query(0, ge=0),
    max_price: float = Query(1000, le=10000)
):
    filtered = [
        item for item in items_db.values()
        if min_price <= item["price"] <= max_price
    ]
    return filtered
```

**3. Flask Equivalent**

```python
# pip install flask

from flask import Flask, request, jsonify, abort
from flask.views import MethodView

app = Flask(__name__)

# In-memory DB
items_db = {}

class ItemAPI(MethodView):
    def get(self, item_id):
        if item_id not in items_db:
            abort(404, description="Item not found")
        return jsonify(items_db[item_id])
    
    def put(self, item_id):
        if item_id not in items_db:
            abort(404)
        data = request.get_json()
        items_db[item_id] = data
        return jsonify({"id": item_id, **data})
    
    def delete(self, item_id):
        if item_id not in items_db:
            abort(404)
        del items_db[item_id]
        return jsonify({"message": "deleted"})

# Register routes
item_view = ItemAPI.as_view('item_api')
app.add_url_rule('/items/<int:item_id>', view_func=item_view, methods=['GET', 'PUT', 'DELETE'])

@app.route('/items/', methods=['POST'])
def create_item():
    data = request.get_json()
    item_id = len(items_db) + 1
    items_db[item_id] = data
    return jsonify({"id": item_id, **data}), 201

if __name__ == "__main__":
    app.run(debug=True)
```

**4. FastAPI Advanced Features**

```python
from fastapi import FastAPI, Depends, Header, HTTPException, status
from fastapi.security import OAuth2PasswordBearer
from pydantic import BaseModel, Field

app = FastAPI()

# Dependency injection
def common_params(q: Optional[str] = None, page: int = 1):
    return {"q": q, "page": page}

@app.get("/items/")
def list_items(params: dict = Depends(common_params)):
    # params contains q and page
    return {"params": params}

# Authentication dependency
oauth2_scheme = OAuth2PasswordBearer(tokenUrl="token")

def get_current_user(token: str = Depends(oauth2_scheme)):
    # Validate token (simplified)
    if token != "secret-token":
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid credentials",
            headers={"WWW-Authenticate": "Bearer"},
        )
    return {"username": "admin"}

@app.get("/users/me")
def read_users_me(current_user: dict = Depends(get_current_user)):
    return current_user

# Response model (filter output)
@app.get("/items/{item_id}", response_model=Item)
def read_item(item_id: int):
    # Return dict akan divalidasi oleh Item model
    return items_db[item_id]

# Status codes
@app.post("/items/", status_code=status.HTTP_201_CREATED)
def create_item(item: Item):
    item_id = len(items_db) + 1
    items_db[item_id] = item.dict()
    return {"id": item_id, **item.dict()}

# Path operation configuration
@app.get("/debug", include_in_schema=False)  # hide from OpenAPI docs
def debug():
    return {"debug": True}
```

**5. Error Handling**

```python
from fastapi import FastAPI, HTTPException, Request
from fastapi.responses import JSONResponse
from fastapi.exceptions import RequestValidationError

app = FastAPI()

# Custom exception handler
@app.exception_handler(HTTPException)
async def http_exception_handler(request: Request, exc: HTTPException):
    return JSONResponse(
        status_code=exc.status_code,
        content={"message": exc.detail, "error_code": exc.status_code}
    )

# Validation error handler (Pydantic)
@app.exception_handler(RequestValidationError)
async def validation_exception_handler(request: Request, exc: RequestValidationError):
    return JSONResponse(
        status_code=400,
        content={"message": "Validation error", "details": exc.errors()}
    )

# Custom exception
class ItemNotFoundError(Exception):
    def __init__(self, item_id: int):
        self.item_id = item_id

@app.exception_handler(ItemNotFoundError)
async def item_not_found_handler(request: Request, exc: ItemNotFoundError):
    return JSONResponse(
        status_code=404,
        content={"message": f"Item {exc.item_id} not found"}
    )
```

**7. WebSockets (FastAPI)**

```python
from fastapi import FastAPI, WebSocket, WebSocketDisconnect

app = FastAPI()

class ConnectionManager:
    def __init__(self):
        self.active_connections: list[WebSocket] = []
    
    async def connect(self, websocket: WebSocket):
        await websocket.accept()
        self.active_connections.append(websocket)
    
    def disconnect(self, websocket: WebSocket):
        self.active_connections.remove(websocket)
    
    async def broadcast(self, message: str):
        for connection in self.active_connections:
            await connection.send_text(message)

manager = ConnectionManager()

@app.websocket("/ws/{client_id}")
async def websocket_endpoint(websocket: WebSocket, client_id: int):
    await manager.connect(websocket)
    try:
        while True:
            data = await websocket.receive_text()
            await manager.broadcast(f"Client {client_id}: {data}")
    except WebSocketDisconnect:
        manager.disconnect(websocket)
```

**9. OpenAPI/Swagger Documentation**

FastAPI auto-generates OpenAPI schema accessible at:
- `/docs` - Swagger UI (interactive)
- `/redoc` - ReDoc (alternative UI)
- `/openapi.json` - raw schema

Customize:
```python
app = FastAPI(
    title="My API",
    description="This is a very awesome API",
    version="1.0.0",
    docs_url="/api/docs",  # customize URL
    redoc_url="/api/redoc"
)

# Custom schema
@app.get("/items/{item_id}", response_model=Item, tags=["items"])
def read_item(
    item_id: int,
    q: Optional[str] = Query(None, description="Query string")
):
    """
    Retrieve item by ID.

    - **item_id**: unique identifier
    - **q**: optional query parameter
    """
    return items_db[item_id]
```

### Soal 1: CRUD API dengan FastAPI

Buat `main.py` dengan FastAPI untuk `Product` resource:

```python
# models.py
from pydantic import BaseModel
from typing import Optional

class Product(BaseModel):
    id: Optional[int] = None
    name: str = Field(..., min_length=1, max_length=100)
    price: float = Field(..., gt=0)
    description: Optional[str] = None
    in_stock: bool = True

# main.py
from fastapi import FastAPI, HTTPException
from models import Product
from typing import List

app = FastAPI()

# In-memory "database"
products_db = {}
next_id = 1

@app.get("/products", response_model=List[Product])
def list_products():
    return list(products_db.values())

@app.get("/products/{product_id}", response_model=Product)
def get_product(product_id: int):
    if product_id not in products_db:
        raise HTTPException(status_code=404, detail="Product not found")
    return products_db[product_id]

@app.post("/products", response_model=Product, status_code=201)
def create_product(product: Product):
    global next_id
    product.id = next_id
    next_id += 1
    products_db[product.id] = product.dict()
    return product

@app.put("/products/{product_id}", response_model=Product)
def update_product(product_id: int, product: Product):
    if product_id not in products_db:
        raise HTTPException(status_code=404, detail="Product not found")
    product.id = product_id
    products_db[product_id] = product.dict()
    return product

@app.delete("/products/{product_id}")
def delete_product(product_id: int):
    if product_id not in products_db:
        raise HTTPException(status_code=404, detail="Product not found")
    del products_db[product_id]
    return {"message": "Deleted"}

# Run: uvicorn main:app --reload
# Docs: http://localhost:8000/docs
```

### Soal 2: Authentication & Protected Endpoints

Tambahkan JWT authentication ke API:

```python
# auth.py
from datetime import datetime, timedelta
from typing import Optional
from jose import JWTError, jwt
from passlib.context import CryptContext

# pip install python-jose[cryptography] passlib[bcrypt]

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")
SECRET_KEY = "your-secret-key-change-in-production"
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 30

class User(BaseModel):
    username: str
    hashed_password: str

# In-memory user DB
users_db = {}

def verify_password(plain_password, hashed_password):
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password):
    return pwd_context.hash(password)

def create_access_token(data: dict, expires_delta: Optional[timedelta] = None):
    to_encode = data.copy()
    expire = datetime.utcnow() + (expires_delta or timedelta(minutes=15))
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

# main.py (extend)
from fastapi import Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer, OAuth2PasswordRequestForm

oauth2_scheme = OAuth2PasswordBearer(tokenUrl="token")

@app.post("/register")
def register(username: str, password: str):
    if username in users_db:
        raise HTTPException(status_code=400, detail="Username taken")
    hashed = get_password_hash(password)
    users_db[username] = {"username": username, "hashed_password": hashed}
    return {"message": "User created"}

@app.post("/token")
def login(form_data: OAuth2PasswordRequestForm = Depends()):
    user = users_db.get(form_data.username)
    if not user or not verify_password(form_data.password, user["hashed_password"]):
        raise HTTPException(status_code=400, detail="Incorrect credentials")
    access_token = create_access_token(data={"sub": user["username"]})
    return {"access_token": access_token, "token_type": "bearer"}

def get_current_user(token: str = Depends(oauth2_scheme)):
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": "Bearer"},
    )
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        username: str = payload.get("sub")
        if username is None:
            raise credentials_exception
    except JWTError:
        raise credentials_exception
    user = users_db.get(username)
    if user is None:
        raise credentials_exception
    return user

@app.get("/users/me")
def read_users_me(current_user: dict = Depends(get_current_user)):
    return current_user

# Protected product endpoints
@app.post("/products", response_model=Product, status_code=201)
def create_product(
    product: Product,
    current_user: dict = Depends(get_current_user)
):
    # Only authenticated users can create
```

### Soal 3: Flask REST API

Implementasi yang sama dengan Flask (tanpa auto-docs):

```python
from flask import Flask, request, jsonify, abort
from flask_httpauth import HTTPTokenAuth
from werkzeug.security import generate_password_hash, check_password_hash

app = Flask(__name__)
auth = HTTPTokenAuth(scheme='Bearer')

tokens = {"secret-token": "admin"}  # token -> user mapping
users = {"admin": generate_password_hash("password")}

@auth.verify_token
def verify_token(token):
    if token in tokens:
        return tokens[token]
    return None

@app.route('/products', methods=['GET'])
def list_products():
    return jsonify(list(products_db.values()))

@app.route('/products/<int:product_id>', methods=['GET'])
def get_product(product_id):
    product = products_db.get(product_id)
    if not product:
        abort(404, description="Product not found")
    return jsonify(product)

@app.route('/products', methods=['POST'])
@auth.login_required
def create_product():
    data = request.get_json()
    if not data or 'name' not in data or 'price' not in data:
        abort(400, description="Invalid request")
    product_id = len(products_db) + 1
    product = {
        'id': product_id,
        'name': data['name'],
        'price': data['price'],
        'description': data.get('description'),
        'in_stock': data.get('in_stock', True)
    }
    products_db[product_id] = product
    return jsonify(product), 201

# ... PUT, DELETE routes

if __name__ == '__main__':
    app.run(debug=True)
```
