## Materi

### Konsep Dasar

Python memiliki tiga jenis methods dalam class:

1. **Instance Method** (default): operates on instance, receives `self`
2. **Class Method**: operates on class itself, receives `cls`, can access class attributes
3. **Static Method**: no `self` or `cls`, pure utility function namespaced dalam class

**Special Methods (Dunder Methods)** adalah methods dengan `__name__` yang memungkinkan class mengimplementasikan behaviors spesifik (seperti `+`, `len()`, iteration, dll).

### Detail Teknis

**1. Class Methods (`@classmethod`)**

```python
class Person:
    population = 0  # class attribute
    
    def __init__(self, name):
        self.name = name
        Person.population += 1
    
    # Instance method
    def greet(self):
        return f"Hello, I'm {self.name}"
    
    # Class method
    @classmethod
    def get_population(cls):
        """Return total population (class-level data)"""
        return f"Total population: {cls.population}"
    
    @classmethod
    def from_full_name(cls, full_name):
        """Alternative constructor (factory method)"""
        # Parse full name, e.g., "Alice Smith" -> name="Alice"
        parts = full_name.split()
        name = parts[0]
        return cls(name)  # create new instance

# Usage
print(Person.get_population())  # "Total population: 0" (class method called on class)
alice = Person("Alice")
print(alice.get_population())   # "Total population: 1" (also works on instance)
print(Person.get_population()) # "Total population: 1"

# Alternative constructor
bob = Person.from_full_name("Bob Johnson")
print(bob.name)  # "Bob"
```

**2. Static Methods (`@staticmethod`)**

```python
class MathUtils:
    @staticmethod
    def add(a, b):
        return a + b
    
    @staticmethod
    def is_even(n):
        return n % 2 == 0
    
    @staticmethod
    def factorial(n):
        if n <= 1:
            return 1
        return n * MathUtils.factorial(n-1)

# Usage - called on class or instance, no self/cls passed
print(MathUtils.add(5, 3))        # 8
print(MathUtils.is_even(4))      # True
print(MathUtils.factorial(5))    # 120

# Also works on instance
m = MathUtils()
print(m.add(2, 2))  # 4
```

**When to use:**
- `@classmethod`: Need access to class (e.g., factory methods, modify class state)
- `@staticmethod`: Utility function yang berhubungan dengan class tapi tidak perlu class/instance state

**3. Special Methods (Dunder Methods)**

```python
class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    # __str__: human-readable (used by print())
    def __str__(self):
        return f"Vector({self.x}, {self.y})"
    
    # __repr__: unambiguous (for debugging)
    def __repr__(self):
        return f"Vector(x={self.x}, y={self.y})"
    
    # Arithmetic operators
    def __add__(self, other):
        if isinstance(other, Vector):
            return Vector(self.x + other.x, self.y + other.y)
        return NotImplemented
    
    def __sub__(self, other):
        if isinstance(other, Vector):
            return Vector(self.x - other.x, self.y - other.y)
        return NotImplemented
    
    def __mul__(self, scalar):
        if isinstance(scalar, (int, float)):
            return Vector(self.x * scalar, self.y * scalar)
        return NotImplemented
    
    # Comparison
    def __eq__(self, other):
        if isinstance(other, Vector):
            return self.x == other.x and self.y == other.y
        return False
    
    def __lt__(self, other):
        # Compare by magnitude
        return (self.x**2 + self.y**2) < (other.x**2 + other.y**2)
    
    # Length
    def __len__(self):
        return 2  # 2D vector
    
    # Indexing
    def __getitem__(self, index):
        if index == 0:
            return self.x
        elif index == 1:
            return self.y
        else:
            raise IndexError("Vector index out of range")
    
    # Iteration (makes object iterable)
    def __iter__(self):
        yield self.x
        yield self.y
    
    # Container methods
    def __contains__(self, value):
        return value == self.x or value == self.y

v1 = Vector(1, 2)
v2 = Vector(3, 4)

print(v1)            # Vector(1, 2)
print(repr(v1))      # Vector(x=1, y=2)
print(v1 + v2)       # Vector(4, 6)
print(v1 * 3)        # Vector(3, 6)
print(v1 == v2)      # False
print(v1 < v2)       # True (magnitude comparison)
print(len(v1))       # 2
print(v1[0])         # 1
print(list(v1))      # [1, 2] (iterable)
print(2 in v1)       # True (contains)
```

**4. Common Special Methods Reference**

```python
# String representation
__str__(self)     # str(obj) -> human-readable
__repr__(self)    # repr(obj) -> unambiguous, eval-able ideally

# Arithmetic operators
__add__(self, other)      # +
__sub__(self, other)      # -
__mul__(self, other)      # *
__truediv__(self, other)  # /
__floordiv__(self, other) # //
__mod__(self, other)      # %
__pow__(self, other)      # **

# Reverse arithmetic (when left operand doesn't support)
__radd__(self, other)     # other + self
__rmul__(self, other)     # other * self
# etc.

# In-place operators (+=, -=, ...)
__iadd__(self, other)     # self += other
# etc.

# Comparison operators
__eq__(self, other)       # ==
__ne__(self, other)       # !=
__lt__(self, other)       # <
__le__(self, other)       # <=
__gt__(self, other)       # >
__ge__(self, other)       # >=

# Container/sequence protocols
__len__(self)             # len(obj)
__getitem__(self, key)    # obj[key]
__setitem__(self, key, value)  # obj[key] = value
__delitem__(self, key)    # del obj[key]
__contains__(self, item)  # item in obj
__iter__(self)            # iterator for obj
__next__(self)            # next item (if iterator)

# Callable
__call__(self, *args, **kwargs)  # obj(*args, **kwargs)

# Attribute access
__getattr__(self, name)   # called when attribute not found
__setattr__(self, name, value)  # called on attribute assignment
__delattr__(self, name)   # called on attribute deletion

# Context manager
__enter__(self)           # with obj as var:
__exit__(self, exc_type, exc_val, exc_tb)  # exit with block

# Numeric conversions
__int__(self)             # int(obj)
__float__(self)           # float(obj)
__complex__(self)         # complex(obj)
__bool__(self)            # bool(obj)
```

**5. Operator Overloading Best Practices**

```python
class Money:
    def __init__(self, amount, currency="USD"):
        self.amount = amount
        self.currency = currency
    
    def __add__(self, other):
        if isinstance(other, Money):
            if self.currency != other.currency:
                raise TypeError("Cannot add different currencies")
            return Money(self.amount + other.amount, self.currency)
        return NotImplemented
    
    def __eq__(self, other):
        if isinstance(other, Money):
            return self.amount == other.amount and self.currency == other.currency
        return False
    
    def __str__(self):
        return f"{self.amount} {self.currency}"
    
    def __repr__(self):
        return f"Money({self.amount}, {self.currency!r})"

m1 = Money(10, "USD")
m2 = Money(20, "USD")
print(m1 + m2)  # 30 USD
```

**6. Emulating Built-in Types**

```python
class Deck:
    """Emulate a list-like deck of cards"""
    def __init__(self, cards):
        self._cards = list(cards)
    
    def __len__(self):
        return len(self._cards)
    
    def __getitem__(self, index):
        return self._cards[index]
    
    def __setitem__(self, index, value):
        self._cards[index] = value
    
    def __delitem__(self, index):
        del self._cards[index]
    
    def __contains__(self, card):
        return card in self._cards
    
    def __iter__(self):
        return iter(self._cards)
    
    def __reversed__(self):
        return reversed(self._cards)

deck = Deck(["A♠", "K♠", "Q♠", "J♠"])
print(len(deck))          # 4
print(deck[0])            # "A♠"
print("K♠" in deck)      # True
for card in deck:
    print(card)
```
