## Materi

### Konsep Dasar

JSON (JavaScript Object Notation) dan CSV (Comma Separated Values) adalah format data yang umum untuk pertukaran data dan storage.

**JSON Types mapping:**
```
Python          JSON
dict            object
list, tuple     array
str             string
int, float      number
True            true
False           false
None            null
```

**CSV:** Flat data, baris = records, kolom = fields, dipisah delimiter (biasanya koma).

### Detail Teknis

**1. JSON Module**

```python
import json

# Python object -> JSON string (serialization)
data = {
    "name": "Alice",
    "age": 25,
    "hobbies": ["reading", "coding"],
    "active": True,
    "balance": 123.45
}

json_str = json.dumps(data)  # serialize ke string
print(json_str)
# {"name": "Alice", "age": 25, "hobbies": ["reading", "coding"], ...}

# Pretty print dengan indentasi
pretty = json.dumps(data, indent=2, sort_keys=True)
print(pretty)
# {
#   "age": 25,
#   "balance": 123.45,
#   "hobbies": ["reading", "coding"],
#   "name": "Alice",
#   "active": true
# }

# Write JSON ke file
with open("data.json", "w", encoding="utf-8") as f:
    json.dump(data, f, indent=2)

# Read JSON dari file
with open("data.json", "r", encoding="utf-8") as f:
    loaded = json.load(f)  # deserialize ke Python object
print(loaded["name"])  # Alice

# JSON string -> Python object
json_str = '{"name": "Bob", "age": 30}'
obj = json.loads(json_str)
```

**2. Custom JSON Encoding**

```python
import json
from datetime import datetime, date

# Default JSON tidak bisa serialize datetime
data = {"timestamp": datetime.now()}
# json.dumps(data)  # TypeError: Object of type datetime is not JSON serializable

# Gunakan `default` parameter untuk custom serialization
def default_serializer(obj):
    if isinstance(obj, (datetime, date)):
        return obj.isoformat()  # convert ke string
    raise TypeError(f"Type {type(obj)} not serializable")

json_str = json.dumps(data, default=default_serializer)

# Atau buat custom JSONEncoder
class MyEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, datetime):
            return obj.isoformat()
        return super().default(obj)

json_str = json.dumps(data, cls=MyEncoder)
```

**3. CSV Module**

```python
import csv

# Write CSV
data = [
    ["name", "age", "city"],
    ["Alice", "25", "Jakarta"],
    ["Bob", "30", "Bandung"],
    ["Charlie", "35", "Surabaya"]
]

with open("people.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.writer(f)
    writer.writerows(data)  # write multiple rows

# Read CSV
with open("people.csv", "r", encoding="utf-8") as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)  # ['name', 'age', 'city'], ['Alice', '25', 'Jakarta'], ...

# DictWriter (more convenient - dictionary to CSV)
people = [
    {"name": "Alice", "age": 25, "city": "Jakarta"},
    {"name": "Bob", "age": 30, "city": "Bandung"},
    {"name": "Charlie", "age": 35, "city": "Surabaya"}
]

fieldnames = ["name", "age", "city"]
with open("people_dict.csv", "w", newline="") as f:
    writer = csv.DictWriter(f, fieldnames=fieldnames)
    writer.writeheader()  # write baris header
    writer.writerows(people)  # write semua dicts

# DictReader (CSV -> list of dicts)
with open("people_dict.csv", "r", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(row["name"], row["age"])  # Alice 25, Bob 30, ...
        # row adalah Dict dengan key = header fields
```

**4. CSV Delimiters & Dialects**

```python
# CSV bisa dengan delimiter selain comma
with open("data.tsv", "r") as f:
    reader = csv.reader(f, delimiter="\t")  # tab-separated

# Atau pipe-separated
with open("data.txt", "r") as f:
    reader = csv.reader(f, delimiter="|")

# Custom dialect
csv.register_dialect(
    "mydialect",
    delimiter=";",
    quotechar='"',
    quoting=csv.QUOTE_MINIMAL,
    lineterminator="\n"
)
with open("data.csv", "r") as f:
    reader = csv.reader(f, dialect="mydialect")
```

**5. JSON Lines Format (NDJSON)**

```python
# JSON Lines: setiap baris adalah JSON object terpisah
# File app.jsonl:
# {"id": 1, "name": "Alice"}
# {"id": 2, "name": "Bob"}

# Read
records = []
with open("data.jsonl", "r", encoding="utf-8") as f:
    for line in f:
        record = json.loads(line.strip())
        records.append(record)

# Write
with open("output.jsonl", "w", encoding="utf-8") as f:
    for record in records:
        f.write(json.dumps(record) + "\n")
```
